>```mathematica
ClearAll["Global`*"];
(*参数设置*)
q1 = 3.0; q2 = -1.0;  (*电荷量*)
pos1 = {-1, 0};       (*正电荷位置*)
pos2 = {1, 0};        (*负电荷位置*)
epsilon0 = 1;         (*真空介电常数*)
(*电势函数*)
potential[x_, y_] := 
  q1/Sqrt[(x - pos1[[1]])^2 + (y - pos1[[2]])^2] + 
   q2/Sqrt[(x - pos2[[1]])^2 + (y - pos2[[2]])^2];
(*电场分量*)
Ex[x_, y_] := 
  q1*(x - pos1[[1]])/((x - pos1[[1]])^2 + (y - pos1[[2]])^2)^(3/2) + 
   q2*(x - pos2[[1]])/((x - pos2[[1]])^2 + (y - pos2[[2]])^2)^(3/2);
Ey[x_, y_] := 
  q1*(y - pos1[[2]])/((x - pos1[[1]])^2 + (y - pos1[[2]])^2)^(3/2) + 
   q2*(y - pos2[[2]])/((x - pos2[[1]])^2 + (y - pos2[[2]])^2)^(3/2);
(*电场强度大小*)
EfieldMag[x_, y_] := Sqrt[Ex[x, y]^2 + Ey[x, y]^2];
(*可视化*)
range = 3;
step = 0.2;
(*电场箭头图*)
fieldPlot = 
  StreamPlot[{Ex[x, y], Ey[x, y]}, {x, -range, range}, {y, -range, range}, StreamPoints -> Fine, StreamScale -> 0.1, 
   StreamColorFunction -> "Rainbow", PlotLabel -> "电偶极子电场分布"];
(*电势等值线图*)
potentialPlot = 
  ContourPlot[potential[x, y], {x, -range, range}, {y, -range, range}, Contours -> 20, ContourShading -> Automatic, ColorFunction -> "TemperatureMap", PlotLabel -> "电偶极子电势分布", Epilog -> {Red, PointSize[0.03], Point[pos1], Blue, Point[pos2]}];
(*组合显示*)
GraphicsRow[{fieldPlot, potentialPlot}, ImageSize -> 800]
